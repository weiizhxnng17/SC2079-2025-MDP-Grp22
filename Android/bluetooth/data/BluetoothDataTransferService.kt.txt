package com.example.mdp_group22.bluetooth.data

import android.bluetooth.BluetoothSocket
import com.example.mdp_group22.bluetooth.domain.BluetoothMessage
import com.example.mdp_group22.bluetooth.domain.TransferFailedException
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.flow.flowOn
import kotlinx.coroutines.withContext
import java.io.IOException

class BluetoothDataTransferService(
    private val socket: BluetoothSocket

) {
    fun listenToIncomingMessages(): Flow<BluetoothMessage> {
        return flow {
            if (!socket.isConnected){
                return@flow
            }
            val buffer = ByteArray(1024)
            while (true) {
                val byeCount = try {
                    socket.inputStream.read(buffer)
                } catch (e: IOException) {
                    throw TransferFailedException()
                }

                emit(
                    buffer.decodeToString(
                        endIndex = byeCount
                    ).toBluetoothMessage(
                        isFromLocalUser = false
                    )
                )
            }
        }.flowOn(Dispatchers.IO)
    }

    suspend fun sendMessage(bytes: ByteArray): Boolean {
        return withContext(Dispatchers.IO) {
            try {
                socket.outputStream.write(bytes)
            } catch (e: IOException) {
                e.printStackTrace()
                return@withContext false
            }

            true
        }
    }
}
